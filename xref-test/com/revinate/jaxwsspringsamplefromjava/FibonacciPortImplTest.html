8   package com.revinate.jaxwsspringsamplefromjava;
9   
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  import static org.junit.jupiter.api.Assertions.assertThrows;
12  
13  import java.lang.reflect.Field;
14  
15  import org.junit.jupiter.api.Test;
16  
17  /**
18   * Tests for {@link FibonacciPortImpl}.
19   */
20  public class FibonacciPortImplTest {
21  
22      @Test
23      void fibonacciDelegatesToNumberServiceForValidIndex() throws Exception {
24          FibonacciPortImpl port = new FibonacciPortImpl();
25          injectNumberService(port, new NumberService());
26  
27          int result = port.fibonacci(5);
28  
29          assertEquals(5, result);
30      }
31  
32      @Test
33      void fibonacciThrowsExceptionForNegativeIndex() throws Exception {
34          FibonacciPortImpl port = new FibonacciPortImpl();
35          injectNumberService(port, new NumberService());
36  
37          FibonacciException exception = assertThrows(FibonacciException.class, () -> port.fibonacci(-1));
38  
39          assertEquals("Index cannot be negative.", exception.getMessage());
40          assertEquals("Index: -1", exception.getDetail());
41      }
42  
43      private void injectNumberService(FibonacciPortImpl port, NumberService service) throws Exception {
44          Field field = FibonacciPortImpl.class.getDeclaredField("numberService");
45          field.setAccessible(true);
46          field.set(port, service);
47      }
48  }
