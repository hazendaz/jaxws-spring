8   package com.revinate.jaxwsspringsample;
9   
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  import static org.junit.jupiter.api.Assertions.assertNotNull;
12  import static org.junit.jupiter.api.Assertions.assertThrows;
13  
14  import com.revinate.sample.service.FactorialFault;
15  
16  import java.lang.reflect.Field;
17  
18  import org.junit.jupiter.api.Test;
19  
20  /**
21   * Tests for {@link FactorialPortImpl}.
22   */
23  public class FactorialPortImplTest {
24  
25      @Test
26      void factorialDelegatesToNumberServiceForValidInput() throws Exception {
27          FactorialPortImpl port = new FactorialPortImpl();
28          injectNumberService(port, new NumberService());
29  
30          int result = port.factorial(5);
31  
32          assertEquals(120, result);
33      }
34  
35      @Test
36      void factorialThrowsFaultForNegativeInput() throws Exception {
37          FactorialPortImpl port = new FactorialPortImpl();
38          injectNumberService(port, new NumberService());
39  
40          FactorialFault fault = assertThrows(FactorialFault.class, () -> port.factorial(-2));
41  
42          assertEquals("Number cannot be negative.", fault.getMessage());
43          assertNotNull(fault.getFaultInfo());
44          assertEquals("Number: -2", fault.getFaultInfo().getFaultInfo());
45      }
46  
47      private void injectNumberService(FactorialPortImpl port, NumberService service) throws Exception {
48          Field field = FactorialPortImpl.class.getDeclaredField("numberService");
49          field.setAccessible(true);
50          field.set(port, service);
51      }
52  }
