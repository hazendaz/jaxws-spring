28  /**
29   * Coverage tests for generated SOAP classes.
30   */
31  public class GeneratedSampleClassesTest {
32  
33      @Test
34      void datatypeAccessorsAndFactoryMethodsWork() {
35          ObjectFactory factory = new ObjectFactory();
36  
37          Fibonacci fibonacci = factory.createFibonacci();
38          fibonacci.setIndex(8);
39          assertEquals(8, fibonacci.getIndex());
40  
41          FibonacciResponse fibonacciResponse = factory.createFibonacciResponse();
42          fibonacciResponse.setReturn(21);
43          assertEquals(21, fibonacciResponse.getReturn());
44  
45          com.revinate.sample.datatype.FibonacciFault fibonacciFault = factory.createFibonacciFault();
46          fibonacciFault.setFaultInfo("bad-index");
47          fibonacciFault.setMessage("invalid");
48          assertEquals("bad-index", fibonacciFault.getFaultInfo());
49          assertEquals("invalid", fibonacciFault.getMessage());
50  
51          Factorial factorial = factory.createFactorial();
52          factorial.setNumber(6);
53          assertEquals(6, factorial.getNumber());
54  
55          FactorialResponse factorialResponse = factory.createFactorialResponse();
56          factorialResponse.setReturn(720);
57          assertEquals(720, factorialResponse.getReturn());
58  
59          com.revinate.sample.datatype.FactorialFault factorialFault = factory.createFactorialFault();
60          factorialFault.setFaultInfo("bad-number");
61          factorialFault.setMessage("invalid");
62          assertEquals("bad-number", factorialFault.getFaultInfo());
63          assertEquals("invalid", factorialFault.getMessage());
64      }
65  
66      @Test
67      void serviceFaultWrappersExposeFaultBeans() {
68          com.revinate.sample.datatype.FibonacciFault fibonacciFaultInfo = new com.revinate.sample.datatype.FibonacciFault();
69          Throwable cause = new IllegalArgumentException("cause");
70  
71          com.revinate.sample.service.FibonacciFault fibonacciFault = new com.revinate.sample.service.FibonacciFault(
72                  "fibonacci-error", fibonacciFaultInfo, cause);
73  
74          assertEquals("fibonacci-error", fibonacciFault.getMessage());
75          assertSame(cause, fibonacciFault.getCause());
76          assertSame(fibonacciFaultInfo, fibonacciFault.getFaultInfo());
77  
78          com.revinate.sample.datatype.FactorialFault factorialFaultInfo = new com.revinate.sample.datatype.FactorialFault();
79          FactorialFault factorialFault = new FactorialFault("factorial-error", factorialFaultInfo, cause);
80  
81          assertEquals("factorial-error", factorialFault.getMessage());
82          assertSame(cause, factorialFault.getCause());
83          assertSame(factorialFaultInfo, factorialFault.getFaultInfo());
84      }
85  
86      @Test
87      void sampleServiceConstructorsAndPortAccessorsAreUsable() throws Exception {
88          SampleService defaultService = new SampleService();
89          assertNotNull(defaultService);
90          assertNotNull(defaultService.getFibonacciPort());
91          assertNotNull(defaultService.getFactorialPort());
92  
93          URL wsdl = getClass().getClassLoader().getResource("sample/wsdl/SampleService.wsdl");
94          assertNotNull(wsdl);
95          QName qName = new QName("http://www.revinate.com/sample", "SampleService");
96  
97          SampleService explicitService = new SampleService(wsdl, qName);
98          assertNotNull(explicitService);
99          assertNotNull(explicitService.getFibonacciPort());
100         assertNotNull(explicitService.getFactorialPort());
101     }
