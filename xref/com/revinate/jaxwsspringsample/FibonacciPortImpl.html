8   package com.revinate.jaxwsspringsample;
9   
10  import com.revinate.sample.service.FibonacciFault;
11  import com.revinate.sample.service.FibonacciPort;
12  
13  import javax.jws.WebService;
14  
15  import org.springframework.beans.factory.annotation.Autowired;
16  import org.springframework.stereotype.Component;
17  
18  @WebService(endpointInterface = "com.revinate.sample.service.FibonacciPort")
19  @Component
20  public class FibonacciPortImpl implements FibonacciPort {
21  
22      @Autowired
23      private NumberService numberService;
24  
25      public int fibonacci(int index) throws FibonacciFault {
26          if (index < 0) {
27              String message = "Index cannot be negative.";
28              com.revinate.sample.datatype.FibonacciFault fault = new com.revinate.sample.datatype.FibonacciFault();
29              fault.setMessage(message);
30              fault.setFaultInfo("Index: " + index);
31              throw new FibonacciFault(message, fault);
32          }
33  
34          return numberService.fibonacci(index);
35      }
36  }
