8   package com.revinate.jaxwsspringsample;
9   
10  import com.revinate.sample.service.FactorialPort;
11  import com.revinate.sample.service.FibonacciPort;
12  import com.revinate.ws.spring.SDDocumentCollector;
13  import com.revinate.ws.spring.SpringService;
14  import com.revinate.ws.spring.internal.SpringBinding;
15  import com.revinate.ws.spring.internal.WSSpringServlet;
16  
17  import java.io.IOException;
18  import java.net.URL;
19  import java.util.Collection;
20  import java.util.Map;
21  
22  import javax.xml.namespace.QName;
23  
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.boot.SpringApplication;
26  import org.springframework.boot.autoconfigure.SpringBootApplication;
27  import org.springframework.boot.web.servlet.ServletRegistrationBean;
28  import org.springframework.context.annotation.Bean;
29  
30  @SpringBootApplication
31  public class Application {
32  
33      private static final Collection<Object> SAMPLESERVICE_METADATA;
34      private static final Object SAMPLESERVICE_PRIMARY_WSDL;
35  
36      static {
37          ClassLoader cl = Application.class.getClassLoader();
38          Map<URL, Object> docs = SDDocumentCollector.collectDocs("sample", cl);
39          SAMPLESERVICE_METADATA = docs.values();
40          SAMPLESERVICE_PRIMARY_WSDL = docs.get(cl.getResource("sample/wsdl/SampleService.wsdl"));
41      }
42  
43      @Autowired
44      private FibonacciPort fibonacciPort;
45  
46      @Autowired
47      private FactorialPort factorialPort;
48  
49      public static void main(String[] args) {
50          SpringApplication.run(Application.class, args);
51      }
52  
53      @Bean
54      WSSpringServlet jaxwsServlet() {
55          return new WSSpringServlet();
56      }
57  
58      @Bean
59      ServletRegistrationBean<WSSpringServlet> jaxwsServletRegistration() {
60          ServletRegistrationBean<WSSpringServlet> bean = new ServletRegistrationBean<>(jaxwsServlet(), "/service/*");
61          bean.setLoadOnStartup(1);
62          return bean;
63      }
64  
65      @Bean
66      SpringService fibonacciService() throws IOException {
67          SpringService service = new SpringService();
68          service.setBean(fibonacciPort);
69          service.setServiceName(new QName("http://www.revinate.com/sample", "SampleService"));
70          service.setPortName(new QName("http://www.revinate.com/sample", "FibonacciPort"));
71          service.setMetadata(SAMPLESERVICE_METADATA);
72          service.setPrimaryWsdl(SAMPLESERVICE_PRIMARY_WSDL);
73          return service;
74      }
75  
76      @Bean
77      SpringService factorialService() throws IOException {
78          SpringService service = new SpringService();
79          service.setBean(factorialPort);
80          service.setServiceName(new QName("http://www.revinate.com/sample", "SampleService"));
81          service.setPortName(new QName("http://www.revinate.com/sample", "FactorialPort"));
82          service.setMetadata(SAMPLESERVICE_METADATA);
83          service.setPrimaryWsdl(SAMPLESERVICE_PRIMARY_WSDL);
84          return service;
85      }
86  
87      @Bean
88      SpringBinding fibonacciBinding() throws Exception {
89          SpringBinding binding = new SpringBinding();
90          binding.setUrl("/service/fibonacci");
91          binding.setService(fibonacciService().getObject());
92          return binding;
93      }
94  
95      @Bean
96      SpringBinding factorialBinding() throws Exception {
97          SpringBinding binding = new SpringBinding();
98          binding.setUrl("/service/factorial");
99          binding.setService(factorialService().getObject());
100         return binding;
101     }
102 }
