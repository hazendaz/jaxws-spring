8   package com.revinate.jaxwsspringsample;
9   
10  import com.revinate.sample.service.FactorialFault;
11  import com.revinate.sample.service.FactorialPort;
12  
13  import javax.jws.WebService;
14  
15  import org.springframework.beans.factory.annotation.Autowired;
16  import org.springframework.stereotype.Component;
17  
18  @WebService(endpointInterface = "com.revinate.sample.service.FactorialPort")
19  @Component
20  public class FactorialPortImpl implements FactorialPort {
21  
22      @Autowired
23      private NumberService numberService;
24  
25      public int factorial(int number) throws FactorialFault {
26          if (number < 0) {
27              String message = "Number cannot be negative.";
28              com.revinate.sample.datatype.FactorialFault fault = new com.revinate.sample.datatype.FactorialFault();
29              fault.setMessage(message);
30              fault.setFaultInfo("Number: " + number);
31              throw new FactorialFault(message, fault);
32          }
33  
34          return numberService.factorial(number);
35      }
36  }
