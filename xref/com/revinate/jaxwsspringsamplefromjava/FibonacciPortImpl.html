8   package com.revinate.jaxwsspringsamplefromjava;
9   
10  import javax.jws.WebService;
11  
12  import org.springframework.beans.factory.annotation.Autowired;
13  import org.springframework.stereotype.Component;
14  
15  @WebService
16  @Component
17  public class FibonacciPortImpl {
18  
19      @Autowired
20      private NumberService numberService;
21  
22      public int fibonacci(int index) throws FibonacciException {
23          if (index < 0) {
24              throw new FibonacciException("Index cannot be negative.", "Index: " + index);
25          }
26  
27          return numberService.fibonacci(index);
28      }
29  }
